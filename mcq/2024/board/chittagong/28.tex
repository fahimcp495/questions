\documentclass[tikz,border=5mm]{standalone}
\usepackage{fontspec}

% Define a specific font command for Bengali only.
% This ensures A, B, C, D use the default English font and don't turn into boxes.
\newfontfamily\benfont{Noto Sans Bengali}[Script=Bengali]

\usetikzlibrary{patterns, decorations.pathreplacing}

\begin{document}

\begin{tikzpicture}[scale=0.6]
    % Define coordinates
    % Let BC = 5 units (proportional to 10 m)
    % Angle C = 60 degrees.
    \coordinate (B) at (0,0);
    \coordinate (C) at (0,5);
    \coordinate (D) at ({-5*tan(60)}, 0); 
    \coordinate (A) at (0, 7.5); 

    % Draw the main triangle
    \draw[line width=1.2pt] (D) -- (B) -- (C) -- cycle;

    % Draw the dashed extension line
    \draw[line width=1.2pt, dashed] (C) -- (A);

    % Draw the Angle Arc at C
    \draw[thick] (C) ++(0,-1.2) arc (270:210:1.2);
    
    % Label the Angle
    \node at (-0.8, 3.8) {60$^\circ$};

    % Draw the Curly Brace
    % We use the standard number '10' and the Bengali font for 'mi'
    \draw[decorate, decoration={brace, amplitude=10pt}, line width=1pt] 
        (C) -- (B) node[midway, right=15pt] {10 \benfont{মি}};

    % Labels for points (Now using standard font)
    \node[right] at (A) {A};
    \node[right] at (C) {C};
    \node[right] at (B) {B};
    \node[left] at (D) {D};

\end{tikzpicture}

\end{document}
