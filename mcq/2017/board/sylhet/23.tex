\draw[thick] (R) -- (Q);

\draw[thick] (R) -- (P);

\draw (Q) ++(0,0.4) -- ++(-0.4,0) -- ++(0,-0.4);

% Draw the angle arc at vertex P (60 degrees)

% Label the vertices
\node[above right] at (P) {P};
\node[below right] at (Q) {Q};
\node[left] at (R) {R};
