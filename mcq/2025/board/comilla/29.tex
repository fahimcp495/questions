\begin{tikzpicture}

% Define coordinates for all points
\coordinate (O) at (0,0);
\coordinate (A) at (-2,0);
\coordinate (B) at (2.5,0);
\coordinate (D) at (0,3);
\coordinate (C) at (2.2,2);

% Draw horizontal line A through O to B
\draw[thick] (A) -- (B);

% Draw vertical line from O to D
\draw[thick] (O) -- (D);

% Draw diagonal line from O to C
\draw[thick] (O) -- (C);

% Draw right angle symbol at O between AB and OD
\draw (0,0.25) -- (0.25,0.25) -- (0.25,0);

% Label point A on the left
\node[left] at (A) {A};

% Label point B on the right
\node[right] at (B) {B};

% Label point O below the origin
\node[below] at (O) {O};

% Label point D at the top
\node[above] at (D) {D};

% Label point C at upper right
\node[above right] at (C) {C};
